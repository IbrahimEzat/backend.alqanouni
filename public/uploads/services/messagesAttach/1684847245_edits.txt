async addTowishlist() {
      let result = await this.sendRequest({
        url: "/wishlist/add",
        dataSend: {
          user_id: this.getToken().tokenable_id,
          course_id: this.id
        }
      });

      if (result.data.status) {
        this.inWishlist = true;
      }
    },
    async checkInWishlist() {
      let result = await this.sendRequest({
        url: "/wishlist/check",
        dataSend: {
          user_id: this.getToken().tokenable_id,
          course_id: this.id
        }
      });

      if (result.data.status) {
        this.inWishlist = true;
      }
    },
    async removeFromWishlist() {
      let result = await this.sendRequest({
        url: "/wishlist/delete",
        dataSend: {
          user_id: this.getToken().tokenable_id,
          course_id: this.id
        }
      });

      if (result.data.status) {
        this.inWishlist = false;
      }
    }
